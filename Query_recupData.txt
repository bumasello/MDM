CREATE OR REPLACE PROCEDURE copiar_dados_entre_bancos AS
  -- Declaração das variáveis para armazenar os dados selecionados
  variavel1 tabela_origem.coluna1%TYPE;
  variavel2 tabela_origem.coluna2%TYPE;
  variavel3 tabela_origem.coluna3%TYPE;
BEGIN
  -- Obter os dados da tabela no banco de origem usando a cláusula SELECT INTO
  -- Substitua 'BANCO_ORIGEM' pelo nome do banco de dados de origem e 'TABELA_ORIGEM' pelo nome da tabela de origem.
  -- Certifique-se de que o database link esteja configurado corretamente para o banco de origem.
  SELECT coluna1, coluna2, coluna3
  INTO variavel1, variavel2, variavel3
  FROM tabela_origem@BANCO_ORIGEM;

  -- Armazenar os dados na tabela de destino no banco atual
  -- Substitua 'TABELA_DESTINO' pelo nome da tabela de destino no banco atual.
  INSERT INTO tabela_destino (coluna1_destino, coluna2_destino, coluna3_destino)
  VALUES (variavel1, variavel2, variavel3);

  COMMIT; -- Comitar as mudanças para o banco de destino
EXCEPTION
  WHEN OTHERS THEN
    ROLLBACK; -- Em caso de erro, desfazer as mudanças
    RAISE;    -- Re-lançar a exceção para tratamento externo, se necessário
END;


CREATE DATABASE LINK nome_do_database_link
CONNECT TO nome_de_usuario IDENTIFIED BY senha
USING 'tns_alias_do_banco_de_destino';
